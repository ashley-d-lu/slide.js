Example 1

HTML:

    <div id='lightbox' class='photos'> 

        <div id='photo1' class='photo'><
            img src='./photography/mountains.jpg'>
        </div>

        <div id='photo2' class='photo'><
            img src='./photography/bridge.jpg'>
        </div>
        
        <div id='photo3' class='photo'><
            img src='./photography/lake.jpg'>
        </div>
    </div>

Relevant CSS:

        .photos {
            height: 300px;
            width: 1230px;
        }
        
        .photo {
            height: 300px;
            width: 400px;
        }

Javascript:

        const lightboxSlider = new Slider('#lightbox', {
            direction: 'horizontal',
            clickToOpen: true,
            keepOriginalOrder: true,
            lightbox: true,
            captions: [
                'Mountains, 2020',
                'Bridge, 2020',
                'Lake, 2020'
            ],
            captionPosition: 'right',
            onlyShowCaptionsOnHover: true,
            scaleFactor: 0.8
        });
        
        lightboxSlider.openSetOnDOMContentLoaded();
